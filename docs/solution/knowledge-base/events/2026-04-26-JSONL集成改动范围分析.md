#### 文件2：新增SopTreeExecutor（决策树执行）

**新增代码**（~200行）：
```python
# backend/kb-service/app/services/sop_tree_executor.py

class SopTreeExecutor:
    """SOP决策树执行引擎（处理JSONL结构）"""
    
    async def execute_entry_checks(
        self,
        decision_tree: dict,
        context: ExecutionContext,
    ) -> str | None:
        """执行entry检查步骤，返回匹配的branch_id"""
        
        flow = decision_tree.get("flow", [])
        
        # 并发执行entry检查（entry-1/2/3可并行）
        for entry_step in flow:
            if not entry_step.get("command_line_executable"):
                continue
            
            # 执行命令
            result = await context.execute(entry_step["command_or_path"])
            
            # 匹配信号
            if self._match_signals(result, entry_step.get("matched_signals")):
                # 返回第一个匹配的branch
                for branch in entry_step.get("branches", []):
                    if self._match_condition(result, branch["when"]):
                        return branch["goto_branch_id"]
        
        return None
    
    async def execute_branch(
        self,
        branch_id: str,
        decision_tree: dict,
        context: ExecutionContext,
    ) -> ExecutionResult:
        """执行branch的checks + solution_steps"""
        
        branches = decision_tree.get("branches", [])
        target_branch = next(b for b in branches if b["branch_id"] == branch_id)
        
        # 串行执行checks
        for check_step in target_branch.get("checks", []):
            if check_step.get("is_high_risk_operation"):
                # 返回pending_confirm，等待用户确认
                return ExecutionResult(
                    status="pending_confirm",
                    risk="high",
                    command=check_step["command_or_path"],
                )
            
            result = await context.execute(check_step["command_or_path"])
            # ... 状态追踪逻辑
        
        return ExecutionResult(status="resolved")
```

- **破坏性**：无（独立服务，不依赖现有代码）
